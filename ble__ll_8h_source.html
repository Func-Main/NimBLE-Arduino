376 /*
377  * CONNECT_REQ
378  * -> InitA (6 bytes)
379  * -> AdvA (6 bytes)
380  * -> LLData (22 bytes)
381  * -> Access address (4 bytes)
382  * -> CRC init (3 bytes)
383  * -> WinSize (1 byte)
384  * -> WinOffset (2 bytes)
385  * -> Interval (2 bytes)
386  * -> Latency (2 bytes)
387  * -> Timeout (2 bytes)
388  * -> Channel Map (5 bytes)
389  * -> Hop Increment (5 bits)
390  * -> SCA (3 bits)
391  *
392  * InitA is the initiators public (TxAdd=0) or random (TxAdd=1) address.
393  * AdvaA is the advertisers public (RxAdd=0) or random (RxAdd=1) address.
394  * LLData contains connection request data.
395  * aa: Link Layer's access address
396  * crc_init: The CRC initialization value used for CRC calculation.
397  * winsize: The transmit window size = winsize * 1.25 msecs
398  * winoffset: The transmit window offset = winoffset * 1.25 msecs
399  * interval: The connection interval = interval * 1.25 msecs.
400  * latency: connection slave latency = latency
401  * timeout: Connection supervision timeout = timeout * 10 msecs.
402  * chanmap: contains channel mapping indicating used and unused data
403  * channels. Only bits that are 1 are usable. LSB is channel 0.
404  * hop_inc: Hop increment used for frequency hopping. Random value in
405  * range of 5 to 16.
406  */
407 #define BLE_CONNECT_REQ_LEN (34)
