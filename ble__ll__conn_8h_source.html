81 /*
82  * Note that the LTK is the key, the SDK is the plain text, and the
83  * session key is the cipher text portion of the encryption block.
84  *
85  * NOTE: we have intentionally violated the specification by making the
86  * transmit and receive packet counters 32-bits as opposed to 39 (as per the
87  * specification). We do this to save code space, ram and calculation time. The
88  * only drawback is that any encrypted connection that sends more than 2^32
89  * packets will suffer a MIC failure and thus be disconnected.
90  */

98  uint64_t host_rand_num;
99  uint8_t iv[8];
