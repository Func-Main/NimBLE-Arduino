214 static inline int ble_ll_phy_to_phy_mode(int phy, int phy_options)
215 {
216  int phy_mode;
217 
218  /*
219  * 'phy' value can be used as 'phy_mode' value unless S=2 coding is explicitly
220  * required. By default we'll use S=2 for Coded.
221  */
222  phy_mode = phy;
223 
224 #if MYNEWT_VAL(BLE_LL_CFG_FEAT_LE_CODED_PHY)
225  if (phy == BLE_PHY_CODED && phy_options == BLE_HCI_LE_PHY_CODED_S2_PREF) {
226  phy_mode = BLE_PHY_MODE_CODED_500KBPS;
227  }
228 #endif
229 
230  return phy_mode;
231 }
