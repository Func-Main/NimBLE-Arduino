360 static inline void sys_memcpy_swap(void *dst, const void *src, size_t length)
361 {
362  __ASSERT(((src < dst && (src + length) <= dst) ||
363  (src > dst && (dst + length) <= src)),
364  "Source and destination buffers must not overlap");
365 
366  src += length - 1;
367 
368  for (; length > 0; length--) {
369  *((u8_t *)dst++) = *((u8_t *)src--);
370  }
371 }
